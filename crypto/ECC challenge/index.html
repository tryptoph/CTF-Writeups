# Step 1: Implement Smart's attack to find private key d
print("\n=== Implementing Smart's attack ===")
print("Smart's attack exploits the weakness in anomalous curves (where #E(Fp) = p)")
print("by lifting the problem to p-adic numbers where discrete log is easy.\n")

def smart_attack(P, Q, p):
    """
    Solve the ECDLP for anomalous curves where #E(F_p) = p
    Returns d such that Q = d*P
    """
    print("1. Lifting curve and points to p-adic numbers...")
    # Construct the curve over Q_p (p-adics)
    E_Qp = EllipticCurve(Qp(p, 2), [a, b])

    # Lift points to p-adics
    print("2. Finding valid p-adic lifts of the points...")
    P_Qp_points = E_Qp.lift_x(ZZ(P.xy()[0]), all=True)
    for point in P_Qp_points:
        if GF(p)(point.xy()[1]) == P.xy()[1]:
            P_lift = point
            break

    Q_Qp_points = E_Qp.lift_x(ZZ(Q.xy()[0]), all=True)
    for point in Q_Qp_points:
        if GF(p)(point.xy()[1]) == Q.xy()[1]:
            Q_lift = point
            break

    print("3. Computing p*P and p*Q in the formal group...")
    # Compute p*P and p*Q in E(Q_p)
    pP = p*P_lift
    pQ = p*Q_lift

    print("4. Extracting the discrete log from the p-adic logarithm...")
    # Extract the discrete log from the p-adic logarithm
    # We're using the fact that in the formal group:
    # log(p*Q) = d*log(p*P)
    x_P, y_P = pP.xy()
    x_Q, y_Q = pQ.xy()

    # The formal logarithm maps to the additive group
    # using the negative of the x/y coordinate
    phi_P = -(x_P/y_P)
    phi_Q = -(x_Q/y_Q)

    # Solve for discrete log: phi_Q = d * phi_P
    d = phi_Q / phi_P

    return ZZ(d)

try:

    d = smart_attack(G, Q, p)

    # Derive the encryption key
    print("\n=== Deriving encryption key ===")
    h = sha256((str(shared_secret[0]) + str(shared_secret[1])).encode())
    key_hex = h.hexdigest()
    key_bytes = bytes.fromhex(key_hex)

    # Decrypt the flag

        # XOR using bitwise operations
        a = int(encrypted_bytes[i])
        b = int(key_bytes[i % len(key_bytes)])
        # Manual XOR implementation: a XOR b
        xor_result = (a | b) - (a & b)
        decrypted_bytes.append(xor_result)

    flag = bytes(decrypted_bytes).decode()
    print(f"\nDecrypted flag: {flag}")

except Exception as e:
    print(f"Attack failed: {e}")
